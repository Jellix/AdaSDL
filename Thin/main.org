* Build/rebuild tags
;; After this (in a ada source file): 
;; `M-.â€™ to find a tag. 
;; `C-u M-.' next alternate.
;; `C-u - M-.' go previous.
(progn
  (defun create-tags (dir-name)
    "Create tags file."
    (interactive "DDirectory: ")
    (eshell-command 
     (format "find %s -type f -name \"*.ad[bs]\" | etags -" dir-name)))
  (create-tags "."))
